ingredients:

  - name: "Sucre"

  - name: "Blancs d'œuf"

  - name: "Amandes entières"
    quantity: 20
    unit: ""

steps:
  - "Mélanger la poudre d’amandes et le sucre."
  - "Ajouter les blancs d’œufs battus en mousse et l’extrait d’amande amère."
  - "Laisser reposer la pâte quelques minutes."
  - "Former des ronds et poser une amande entière dessus."
  - "Cuire au four à 150°C (très peu cuit)."
